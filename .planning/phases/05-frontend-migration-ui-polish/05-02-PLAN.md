**1. Add mapContentItem() normalizer function.**

Place this near the top of the non-module script, after the global variable declarations (around line 2340):

```javascript
/**
 * Normalizes a Supabase content_items row to the field format
 * that all rendering functions (renderCalendar, renderScheduleGrid,
 * openScheduleCard, updateStats, etc.) expect.
 *
 * DB schema fields -> Frontend expected fields:
 * - scheduled_date -> date
 * - content -> post_text (and also keep as content)
 * - theme_id -> theme (joined theme name, or 'General' fallback)
 * - status: pending_review -> pending (for frontend compatibility)
 * - day_number: computed from scheduled_date (day of month)
 * - row_number: not needed (Sheets artifact), omit
 * - hashtags: extracted from content if present, or empty string
 */
function mapContentItem(dbRow, themeMap) {
  const scheduledDate = dbRow.scheduled_date || '';
  const dayNumber = scheduledDate ? new Date(scheduledDate + 'T00:00:00').getDate() : 0;

  // Normalize status: DB uses 'pending_review', frontend checks 'pending'
  let status = dbRow.status;
  if (status === 'pending_review') status = 'pending';

  // Get theme name from join or themeMap lookup
  let themeName = 'General';
  if (dbRow.themes && dbRow.themes.name) {
    themeName = dbRow.themes.name;
  } else if (themeMap && dbRow.theme_id && themeMap[dbRow.theme_id]) {
    themeName = themeMap[dbRow.theme_id];
  }

  // Extract hashtags from content if present (look for # patterns at end)
  let hashtags = '';
  const contentText = dbRow.content || '';
  const hashtagMatch = contentText.match(/((?:#\w+\s*){2,})$/);
  if (hashtagMatch) {
    hashtags = hashtagMatch[1].trim();
  }

  return {
    id: dbRow.id,
    day_number: dayNumber,
    date: scheduledDate,
    platform: dbRow.platform || '',
    theme: themeName,
    content_type: dbRow.content_type || 'text',
    post_text: contentText,
    content: contentText,
    hashtags: hashtags,
    image_url: dbRow.image_url || '',
    video_url: dbRow.video_url || '',
    status: status,
    title: dbRow.title || '',
    image_prompt: dbRow.image_prompt || '',
    feedback: dbRow.feedback || '',
    scheduled_time: dbRow.scheduled_time || '',
    campaign_id: dbRow.campaign_id || null,
    theme_id: dbRow.theme_id || null
  };
}
```

**2. Replace fetchCalendarData() function.**

Find the existing `fetchCalendarData()` function (it calls `authenticatedFetch` to an n8n webhook like `eluxr-approval-list`). Replace the entire function with:

```javascript
async function fetchCalendarData() {
  try {
    // Fetch content items directly from Supabase (RLS filters by user_id)
    const { data: items, error } = await window.supabase
      .from('content_items')
      .select('*, themes(name)')
      .order('scheduled_date', { ascending: true });

    if (error) {
      console.error('Error fetching content from Supabase:', error);
      return { all: [], approved: [], pending: [], rejected: [] };
    }

    if (!items || items.length === 0) {
      return { all: [], approved: [], pending: [], rejected: [] };
    }

    // Normalize all items through the mapper
    const mapped = items.map(item => mapContentItem(item, null));

    // Categorize by status
    const approved = mapped.filter(i => i.status === 'approved');
    const pending = mapped.filter(i => i.status === 'pending');
    const rejected = mapped.filter(i => i.status === 'rejected');

    return { all: mapped, approved, pending, rejected };
  } catch (err) {
    console.error('fetchCalendarData error:', err);
    return { all: [], approved: [], pending: [], rejected: [] };
  }
}
```

    // Also fetch content items for the schedule grid
    await fetchAndDisplayCalendar(true);

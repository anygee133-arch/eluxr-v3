<objective>
Add the ICP (Ideal Customer Profile) card to the Setup tab that loads from Supabase on login, displays structured sections (summary, audience, pain points, messaging, hashtags), shows skeleton loading state, refreshes after pipeline step 1, and gracefully degrades when JSONB fields are null.

**1. Add loadICP() function:**

```javascript
let icpLoaded = false;

async function loadICP() {
  const icpCard = document.getElementById('icp-card');
  const icpSkeleton = document.getElementById('icp-skeleton');
  const icpContent = document.getElementById('icp-content');

  if (!icpCard) return;

  try {
    const { data: icp, error } = await window.supabase
      .from('icps')
      .select('*')
      .single();

    if (error && error.code !== 'PGRST116') {
      // PGRST116 = no rows returned (not an error for us)
      console.error('Error loading ICP:', error);
      return;
    }

    if (!icp) {
      // No ICP exists yet -- hide the card
      icpCard.style.display = 'none';
      return;
    }

    // Show the card and render content
    icpCard.style.display = 'block';
    if (icpSkeleton) icpSkeleton.style.display = 'none';
    if (icpContent) icpContent.style.display = 'block';

    renderICPCard(icp);
    icpLoaded = true;

  } catch (err) {
    console.error('loadICP error:', err);
  }
}

function showICPSkeleton() {
  const icpCard = document.getElementById('icp-card');
  const icpSkeleton = document.getElementById('icp-skeleton');
  const icpContent = document.getElementById('icp-content');

  if (icpCard) icpCard.style.display = 'block';
  if (icpSkeleton) icpSkeleton.style.display = 'block';
  if (icpContent) icpContent.style.display = 'none';
}

function renderICPCard(icp) {
  // Summary section (always show -- primary fallback)
  const summarySection = document.getElementById('icp-summary-section');
  const summaryText = document.getElementById('icp-summary-text');
  if (summaryText) {
    const summary = icp.icp_summary || '';
    if (summary) {
      summaryText.innerHTML = formatICPText(summary);
      if (summarySection) summarySection.style.display = 'block';
    }
  }

  // Target Audience (from demographics JSONB)
  renderICPSection('icp-audience-section', 'icp-audience-text', icp.demographics);

  // Pain Points (from psychographics JSONB)
  renderICPSection('icp-painpoints-section', 'icp-painpoints-text', icp.psychographics);

  // Content Opportunities (from content_opportunities JSONB)
  renderICPSection('icp-messaging-section', 'icp-messaging-text', icp.content_opportunities);

  // Recommended Hashtags (from recommended_hashtags JSONB)
  const hashtagsSection = document.getElementById('icp-hashtags-section');
  const hashtagsText = document.getElementById('icp-hashtags-text');
  if (hashtagsText && icp.recommended_hashtags) {
    const hashtags = icp.recommended_hashtags;
    if (Array.isArray(hashtags) && hashtags.length > 0) {
      hashtagsText.innerHTML = hashtags.map(h =>
        `<span style="display: inline-block; background: rgba(22, 163, 74, 0.15); color: #16a34a; padding: 2px 10px; border-radius: 12px; font-size: 13px; margin: 2px 4px 2px 0;">${h.startsWith('#') ? h : '#' + h}</span>`
      ).join('');
      if (hashtagsSection) hashtagsSection.style.display = 'block';
    } else if (typeof hashtags === 'object' && !Array.isArray(hashtags)) {
      // Handle object format
      const tags = Object.values(hashtags).flat().filter(Boolean);
      if (tags.length > 0) {
        hashtagsText.innerHTML = tags.map(h =>
          `<span style="display: inline-block; background: rgba(22, 163, 74, 0.15); color: #16a34a; padding: 2px 10px; border-radius: 12px; font-size: 13px; margin: 2px 4px 2px 0;">${h.startsWith('#') ? h : '#' + h}</span>`
        ).join('');
        if (hashtagsSection) hashtagsSection.style.display = 'block';
      }
    }
  }

  // Last Analyzed timestamp
  const lastAnalyzed = document.getElementById('icp-last-analyzed');
  if (lastAnalyzed && icp.updated_at) {
    const date = new Date(icp.updated_at);
    lastAnalyzed.textContent = `Last analyzed: ${date.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })}`;
  }
}

function renderICPSection(sectionId, textId, jsonbData) {
  const section = document.getElementById(sectionId);
  const textEl = document.getElementById(textId);

  if (!textEl || !jsonbData) return;

  let html = '';

  if (typeof jsonbData === 'string') {
    html = formatICPText(jsonbData);
  } else if (Array.isArray(jsonbData)) {
    if (jsonbData.length > 0) {
      html = '<ul>' + jsonbData.map(item => `<li>${escapeHTML(typeof item === 'string' ? item : JSON.stringify(item))}</li>`).join('') + '</ul>';
    }
  } else if (typeof jsonbData === 'object') {
    // Object with key-value pairs -- render as labeled list
    const entries = Object.entries(jsonbData).filter(([k, v]) => v != null && v !== '');
    if (entries.length > 0) {
      html = entries.map(([key, value]) => {
        const label = key.replace(/_/g, ' ').replace(/\b\w/g, l => l.toUpperCase());
        const val = Array.isArray(value) ? value.join(', ') : String(value);
        return `<div style="margin-bottom: 6px;"><strong style="color: var(--text-primary, #e2e8f0);">${escapeHTML(label)}:</strong> ${escapeHTML(val)}</div>`;
      }).join('');
    }
  }

  if (html) {
    textEl.innerHTML = html;
    if (section) section.style.display = 'block';
  }
}

function formatICPText(text) {
  // Convert plain text to HTML with line breaks and basic formatting
  return escapeHTML(text).replace(/\n/g, '<br>');
}

function escapeHTML(str) {
  const div = document.createElement('div');
  div.appendChild(document.createTextNode(str));
  return div.innerHTML;
}
```

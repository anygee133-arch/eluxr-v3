```javascript
async function generateContent(formData) {
  // Disable generate button to prevent duplicate runs
  const generateBtn = document.querySelector('.btn-generate, [onclick*="handleFormSubmit"]');
  if (generateBtn) generateBtn.disabled = true;

  try {
    // Single call to orchestrator -- returns 202 immediately
    const response = await authenticatedFetch(`${API_BASE}/eluxr-generate-content`, {
      method: 'POST',
      body: JSON.stringify(formData)
    });

    if (response.status === 202) {
      const data = await response.json();
      const runId = data.pipeline_run_id;

      if (!runId) {
        showToast('Failed to start pipeline: no run ID returned', 'error');
        if (generateBtn) generateBtn.disabled = false;
        return;
      }

      // Store run ID for recovery
      currentPipelineRunId = runId;
      localStorage.setItem('eluxr_active_pipeline_run', runId);

      // Show progress UI and navigate to phase 2
      goToPhase(2);
      resetProgressUI();

      // Subscribe to real-time progress
      subscribeToProgress(runId);

      if (data.resumed) {
        showToast('Resuming existing pipeline run...', 'info');
      }
    } else {
      const errorData = await response.json().catch(() => ({}));
      showToast(`Failed to start generation: ${errorData.error || response.statusText}`, 'error');
      if (generateBtn) generateBtn.disabled = false;
    }
  } catch (error) {
    console.error('Pipeline trigger failed:', error);
    showToast('Failed to start content generation. Please try again.', 'error');
    if (generateBtn) generateBtn.disabled = false;
  }
}
```

**3. Replace `updateProgressUI()` function (around line 3373-3397) with Realtime-driven functions:**

```javascript
// === REAL-TIME PROGRESS TRACKING ===

let progressChannel = null;
let currentPipelineRunId = null;

const STEP_LABELS = [
  'Analyzing your business...',
  'Creating content themes...',
  'Writing posts...',
  'Generating images...',
  'Creating videos...',
  'Syncing calendar...'
];

function resetProgressUI() {
  document.getElementById('progress-bar').style.width = '0%';
  document.getElementById('progress-text').textContent = '0%';
  document.getElementById('phase-text').textContent = 'Starting pipeline...';

  document.querySelectorAll('#phase-2 .phase-item').forEach((item) => {
    item.classList.remove('active', 'completed');
    const icon = item.querySelector('.phase-icon');
    const step = item.getAttribute('data-step');
    if (icon) icon.textContent = step;
  });
}

function subscribeToProgress(runId) {
  // Clean up any existing subscription
  if (progressChannel) {
    window.supabase.removeChannel(progressChannel);
    progressChannel = null;
  }

  progressChannel = window.supabase
    .channel(`pipeline-${runId}`)
    .on(
      'postgres_changes',
      {
        event: 'UPDATE',
        schema: 'public',
        table: 'pipeline_runs',
        filter: `id=eq.${runId}`,
      },
      (payload) => {
        const run = payload.new;
        renderProgress(run);
      }
    )
    .subscribe((status) => {
      if (status === 'SUBSCRIBED') {
        console.log('Subscribed to pipeline progress:', runId);
      } else if (status === 'CHANNEL_ERROR') {
        console.error('Realtime subscription error for pipeline:', runId);
        showToast('Lost connection to progress updates. Refresh to check status.', 'error');
      }
    });
}

function renderProgress(run) {
  const currentStep = run.current_step || 0;
  const totalSteps = run.total_steps || 6;
  const percentage = Math.round((currentStep / totalSteps) * 100);

  // Update progress bar
  document.getElementById('progress-bar').style.width = `${percentage}%`;
  document.getElementById('progress-text').textContent = `${percentage}%`;
  document.getElementById('phase-text').textContent = run.step_label || STEP_LABELS[currentStep - 1] || 'Processing...';

  // Update step list checkmarks
  document.querySelectorAll('#phase-2 .phase-item').forEach((item) => {
    const stepNum = parseInt(item.getAttribute('data-step'));
    item.classList.remove('active', 'completed');
    const icon = item.querySelector('.phase-icon');

    if (stepNum < currentStep) {
      // Completed step
      item.classList.add('completed');
      if (icon) icon.textContent = '\u2713';
    } else if (stepNum === currentStep) {
      // Active step
      item.classList.add('active');
      if (icon) icon.textContent = stepNum;
    } else {
      // Future step
      if (icon) icon.textContent = stepNum;
    }
  });

  // Handle completion
  if (run.status === 'completed') {
    cleanupProgress();
    showToast('Content generation complete!', 'success');
    // Re-enable generate button
    const generateBtn = document.querySelector('.btn-generate, [onclick*="handleFormSubmit"]');
    if (generateBtn) generateBtn.disabled = false;
    // Fetch and display the generated content
    fetchAndDisplayCalendar();
    // Navigate to calendar tab after a brief delay to show 100%
    setTimeout(() => goToPhase(3), 1500);
  } else if (run.status === 'failed') {
    cleanupProgress();
    showToast(`Generation failed: ${run.error_message || 'Unknown error'}`, 'error');
    const generateBtn = document.querySelector('.btn-generate, [onclick*="handleFormSubmit"]');
    if (generateBtn) generateBtn.disabled = false;
  }
}

function cleanupProgress() {
  if (progressChannel) {
    window.supabase.removeChannel(progressChannel);
    progressChannel = null;
  }
  currentPipelineRunId = null;
  localStorage.removeItem('eluxr_active_pipeline_run');
}
```

**4. Add page refresh recovery function:**

```javascript
async function checkActivePipeline() {
  try {
    const { data: runs, error } = await window.supabase
      .from('pipeline_runs')
      .select('*')
      .eq('status', 'running')
      .order('created_at', { ascending: false })
      .limit(1);

    if (error || !runs || runs.length === 0) return;

    const activeRun = runs[0];

    // Check if run is stale (> 15 minutes)
    const startedAt = new Date(activeRun.started_at || activeRun.created_at);
    const minutesElapsed = (Date.now() - startedAt.getTime()) / 60000;

    if (minutesElapsed > 15) {
      // Stale run -- don't restore, just warn
      console.warn('Stale pipeline run detected:', activeRun.id, `(${Math.round(minutesElapsed)} min old)`);
      showToast('A previous generation may have timed out. You can start a new one.', 'error');
      return;
    }

    // Active run found -- restore progress UI
    currentPipelineRunId = activeRun.id;
    localStorage.setItem('eluxr_active_pipeline_run', activeRun.id);

    // Navigate to progress view and restore state
    goToPhase(2);
    renderProgress(activeRun);
    subscribeToProgress(activeRun.id);

    // Disable generate button
    const generateBtn = document.querySelector('.btn-generate, [onclick*="handleFormSubmit"]');
    if (generateBtn) generateBtn.disabled = true;

    showToast('Resuming active pipeline...', 'info');
  } catch (err) {
    console.error('Error checking active pipeline:', err);
  }
}
```

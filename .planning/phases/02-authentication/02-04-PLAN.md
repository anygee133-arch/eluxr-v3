/**
 * Wrapper for all n8n webhook requests. Attaches JWT and handles auth errors.
 * @param {string} url - The webhook URL
 * @param {object} options - Standard fetch options (method, headers, body)
 * @returns {Promise<Response>} - The fetch response
 */
async function authenticatedFetch(url, options = {}) {
  // Get current session
  const { data: { session } } = await supabase.auth.getSession()

  if (!session) {
    showLoginPage()
    throw new Error('Not authenticated')
  }

  const headers = {
    'Content-Type': 'application/json',
    ...(options.headers || {})
  }

  let fetchOptions = { ...options, headers }

  // JWT delivery based on CORS test result
  // IMPORTANT: Read CORS_RESULT from 02-01-SUMMARY.md and use the correct approach

  // If CORS_RESULT = "header":
  // headers['Authorization'] = `Bearer ${session.access_token}`
  // fetchOptions = { ...options, headers }

  // If CORS_RESULT = "body":
  // let body = options.body ? JSON.parse(options.body) : {}
  // body._auth_token = session.access_token
  // fetchOptions = { ...options, headers, body: JSON.stringify(body) }

  // === IMPLEMENT THE CORRECT APPROACH BASED ON 02-01-SUMMARY ===
  // Delete the unused approach and keep the correct one.

  try {
    const response = await fetch(url, fetchOptions)

    if (response.status === 401) {
      // Token rejected -- try refreshing
      console.log('Got 401, attempting token refresh...')
      const { data: { session: newSession } } = await supabase.auth.refreshSession()

      if (!newSession) {
        console.log('Token refresh failed, redirecting to login')
        showLoginPage()
        throw new Error('Session expired')

      throw error
    }
    throw error
  }
}

// Expose to window for non-module scripts

**Node 1: Execute Sub-Workflow**
- Workflow ID: {{ AUTH_VALIDATOR_WORKFLOW_ID from 02-02-SUMMARY }}
- Input: Pass headers, body, and query from the webhook

- Condition: `{{ $json.authenticated }}` equals `true`
- True path: continues to existing business logic, with `user_id` available via `{{ $json.user_id }}`
- False path: goes to 401 response

**Node 3: Respond to Webhook (401)**
- Only on the False path
- Response Code: 401
- Body: `{ "error": "{{ $json.error }}", "success": false }`
